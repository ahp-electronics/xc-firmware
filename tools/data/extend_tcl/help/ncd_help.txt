[ncd]
NCD extended Tcl commands
For more information on a specific command, type the command without any options:
ncd_port          NCD port command
ncd_inst          NCD instance command
ncd_net           NCD net command
ncd_attr          NCD attribute command

[ncd_port]
ncd_port {get|export} [options]
   get [-in] [-out] [-bidi] 
                  Returns list of port tcl objects in a comma-separated 
                  string in following format:
                  port_name1,port_name2,port_name3
                  An empty or nil string is returned if there is no port. If no 
                  option is specified then all ports will be returned. If more 
                  than one option is specified then a combination of results 
                  from each option is returned.
     -in          Input ports
     -out         Output ports
     -bidi        Bidirectional ports
   export {-csv|-lpf} filename [filePath]         
                  Exports pin configuration to userspecified file in either 
                  CSV or LPF format.
                  CSV formation is generated by exporting Device Pin Layout 
                  and LPF format is generated through NCD backannotation.
                  If filePath is not specified then ./ or current directory 
                  is the default.
     -csv filename [filePath]
     -lpf filename [filePath]
   export -pinout filename [filePath]
                  Exports pinouts from command-line to userspecified file in 
                  CSV format.
                  If filePath is not specified then ./ or current directory 
                  is the default.

[ncd_inst]
ncd_inst get [options]
   get [-slice] [-dcs] [-ebr] [-dsp] [-pio]
                  Returns list of hierarchical instance name in a comma-separated
                  string in following format:
                  hier_inst_name1,hier_inst_name2,hier_inst_name3
                  An empty or nil string is returned if there is no port. If no 
                  option is specified then all ports will be returned. If more 
                  than one option is specified then a combination of results from 
                  each option is returned.
     -slice
     -dcs
     -ebr
     -dsp
     -pio

[ncd_net]
ncd_net get -inst hierInstName [options]
   get -inst hierInstName [-pin index | -fanin | -fanout]
                  Returns a list of net/signal tcl objects in a comma-separated 
                  string in following format:
                  hier_net_name1,hier_net_name2,hier_net_name3
                  An empty or nil string is returned if there is no net.
     -inst hierInstName
     -pin  index
     -fanin
     -fanout

[ncd_attr]
ncd_attr get {-inst|-net|-port} name {options}
   get -inst hierInstName -attributeName
                  Retuns a list of attributes for specified instances.
                  Acceptable attributeName are:
                      pincount
                      fanincount
                      fanoutcount
                      type: possible returned values are
                          slice, dcs, ebr, dsp, pio
                      placed: possible returned values are
                          yes, no
   get -net netName -attributeName
                  Returns a list of attributes for specified nets.
                  Acceptable attributeName are:
	              fanincount
                      fanoutcount
	              routed: possible returned values are
	                  yes, no, partial
   get -port portName -attributeName
                  Returns a list of attributes for specified port.
	          Acceptable attributeName are:
                      direction: possible returned values are 
                          in, out, bidi
                      configuration [key]:
                          List of supported configurations key, i.e. PCICLAMP, 
                          IO_TYPE, PULLMODE, SLEWRATE, OPENDRAIN; are device
                          dependent. Please refers to spreadsheet view document
                          of on-line help for more information.
